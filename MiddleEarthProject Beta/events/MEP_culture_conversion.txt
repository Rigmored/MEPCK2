#### For Men and Hobbits ####

# One of your province is under your rule since a long time.
character_event = {
	id = cultureconversion.0
	desc = "EVTDESCcultureconversion.0"
	picture = GFX_evt_throne_room
	
	trigger = {
		any_realm_province = {
			NOT = { culture = ROOT }
			NOT = { religion = ROOT }
			NOT = { culture_group = culture_group_wilderness }
			NOT = { has_province_flag = local_culture }
		}
		NOT = { culture_group = culture_group_melkor } ## Specific events
		NOT = { culture_group = culture_group_eru }
		NOT = { culture_group = culture_group_dwarves }
		NOT = { culture_group = culture_group_quendi }
		NOT = { culture_group = culture_group_amanyar }
		NOT = { culture_group = culture_group_umanyar }
		NOT = { culture_group = culture_group_moriquendi }
		NOT = { culture_group = culture_group_onodrim }
		independent = yes
		NOT = { has_character_flag = provinceconversion }
		war = no
	}
	
	mean_time_to_happen = {
		months = 1200 ### 100 years
		
		modifier = {
			factor = 0.95
			prestige = 2500
		}
		modifier = {
			factor = 0.90
			prestige = 4500
		}
		modifier = {
			factor = 0.85
			prestige = 6500
		}
		modifier = {
			factor = 0.80
			prestige = 10000
		}		
	}
	

	option = {
		name = "EVTOPTAcultureconversion.0" ## This land is ours now.
		character_event = { id = cultureconversion.1 days = 10 }
		random_realm_province = {
			limit = {
				NOT = { culture = ROOT }
				NOT = { religion = ROOT }
				NOT = { culture_group = culture_group_wilderness }
				NOT = { has_province_flag = local_culture }
			}
			set_province_flag = culture_conversion
			custom_tooltip = { text = EVTTOOLTIPcultureconversion.1 }
		}
		set_character_flag = provinceconversion
	}
	
	option = {
		name = "EVTOPTBcultureconversion.0" ## I prefer to keep the local inhabitants of this province
		trigger = { ai = no }
		random_realm_province = {
			limit = {
				NOT = { culture = ROOT }
				NOT = { religion = ROOT }
				NOT = { has_province_flag = local_culture }
			}
			set_province_flag = local_culture
		}		
	}	
}

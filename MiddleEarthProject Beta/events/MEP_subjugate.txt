namespace = subjugate


############# Subjugate another realm #############
####### Your emissary is in the other realm capital #######
character_event = {
	id = subjugate.0
	title = "EVTTITsubjugate.0"
	desc = "EVTDESCsubjugate.0"
	picture = "GFX_evil"

	trigger = {
		NOT = { has_character_modifier = subjugate_realm }
		age = 1
		trait = evil_side
		job_chancellor = {
			has_job_action = action_subjugate
			location = {
				any_province_lord = {
					NOT = { is_liege_of = ROOT }
					NOT = { character = ROOT }
					NOT = { same_realm = ROOT }
					independent = yes
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}	
	
	immediate = {
		ROOT = { add_character_modifier = { name = subjugate_realm duration = 1460 } }
	}
	
	option = {
		name = "EVTOPTAsubjugate.0" ## Your emissary will ask for wealth and slaves
		ai_chance = {
			factor = 70
		}
		ROOT = {
			job_chancellor = {
				location = {
					any_province_lord = {
						limit = {
							NOT = { is_liege_of = ROOT }
							NOT = { character = ROOT }
							NOT = { same_realm = ROOT }
							independent = yes
						}
						narrative_event = { id = subjugate.1 days = 20 }
					}
				}
			}
		}		
	}
	
	option = {
		name = "EVTOPTBsubjugate.0" ## Your emissary will ask them to join the crown.
		ai_chance = {
			factor = 20
		}		
		trigger = {
			OR = {
				prestige = 800
				trait = great_orc
				trait = maiar
				trait = nazgul
			}		
		}
		prestige = -100
		wealth = -200		
		ROOT = {
			job_chancellor = {
				location = {
					any_province_lord = {
						limit = {
							NOT = { is_liege_of = ROOT }
							NOT = { character = ROOT }
							NOT = { same_realm = ROOT }
							independent = yes
						}
						narrative_event = { id = subjugate.4 days = 20 }
					}
				}
			}
		}		
	}	
	
	option = {
		name = "EVTOPTCsubjugate.0" ## Your emissary won't do anything.	
		ai_chance = {
			factor = 10
		}			
	}		
}

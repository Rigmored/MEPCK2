		trait = faqih
	}	
		modifier = {
		factor = 0.5 # Increases chances by ten
		trait = ambitious
	}
		modifier = {
		factor = 0.5 # Increases chances by ten
		trait = scholar
	}	
	
}
	option = {
		name = gnosis_objective.1a
		trigger = { not = {trait = school_gnosis_1}}

		}
}
